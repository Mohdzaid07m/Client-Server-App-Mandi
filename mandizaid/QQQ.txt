     SELECT      P_DATE   ,    SENDER_NO ,  SNAME ,    TRUCK_NO ,   SEND_DATE ,   ITEM_CODE ,   CATEG ,    SUM(QTY)  ,     ROUND(RATE) ,  SUM(TOTAL)
     FROM BUYER_DETAIL1 WHERE P_DATE=&TD GROUP BY    P_DATE   ,    SENDER_NO ,  SNAME ,    TRUCK_NO ,   SEND_DATE ,   ITEM_CODE ,   CATEG , ROUND(RATE);
